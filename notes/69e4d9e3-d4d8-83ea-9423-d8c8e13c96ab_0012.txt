FIX 3 — Use Mask-Based Recoloring (Correct Method)
Working JS (Core Pattern)

function recolorBalloonRegions(ctx, image, masks, colors) {
    ctx.drawImage(image, 0, 0);

    masks.forEach((mask, i) => {
        ctx.save();

        // Apply mask
        ctx.globalCompositeOperation = 'destination-in';
        ctx.drawImage(mask, 0, 0);

        // Apply color
        ctx.globalCompositeOperation = 'multiply';
        ctx.fillStyle = colors[i % colors.length];
        ctx.fillRect(0, 0, ctx.canvas.width, ctx.canvas.height);

        // Restore highlights
        ctx.globalCompositeOperation = 'screen';
        ctx.globalAlpha = 0.2;
        ctx.fillStyle = '#ffffff';
        ctx.fillRect(0, 0, ctx.canvas.width, ctx.canvas.height);

        ctx.restore();
    });
}

Why this works

    Keeps real photo

    Changes only balloon regions

    Preserves realism

3. Recoloring Engine (Correct Implementation)

function recolor(ctx, img, regions, colors) {
    ctx.drawImage(img, 0, 0);

    regions.forEach((region, i) => {
        const color = colors[i % colors.length];

        ctx.save();

        // Clip to balloon region
        ctx.beginPath();
        region.path(ctx); // region defines its own shape
        ctx.clip();

        // Apply color (preserves luminance)
        ctx.globalCompositeOperation = 'color';
        ctx.fillStyle = color;
        ctx.fillRect(0, 0, ctx.canvas.width, ctx.canvas.height);

        // Restore shading
        ctx.globalCompositeOperation = 'multiply';
        ctx.globalAlpha = 0.4;
        ctx.drawImage(img, 0, 0);

        // Restore highlights
        ctx.globalCompositeOperation = 'screen';
        ctx.globalAlpha = 0.15;
        ctx.fillStyle = '#ffffff';
        ctx.fillRect(0, 0, ctx.canvas.width, ctx.canvas.height);

        ctx.restore();
    });
}
